-- Add a task quickly
CREATE OR REPLACE PROCEDURE add_task (
  p_user_id    IN NUMBER,
  p_title      IN VARCHAR2,
  p_desc       IN VARCHAR2,
  p_category_id IN NUMBER,
  p_due_date   IN DATE,
  p_status     IN VARCHAR2 DEFAULT 'pending'
) AS
BEGIN
  INSERT INTO tasks (user_id, title, description, category_id, due_date, status)
  VALUES (p_user_id, p_title, p_desc, p_category_id, p_due_date, p_status);
END;
/
-- Add an expense quickly
CREATE OR REPLACE PROCEDURE add_expense (
  p_user_id     IN NUMBER,
  p_category_id IN NUMBER,
  p_task_id     IN NUMBER,
  p_amount      IN NUMBER,
  p_desc        IN VARCHAR2,
  p_exp_date    IN DATE
) AS
BEGIN
  INSERT INTO expenses (user_id, category_id, task_id, amount, description, expense_date)
  VALUES (p_user_id, p_category_id, p_task_id, p_amount, p_desc, p_exp_date);
END;
/
-- Update task status
CREATE OR REPLACE PROCEDURE set_task_status (
  p_task_id IN NUMBER,
  p_status  IN VARCHAR2
) AS
BEGIN
  UPDATE tasks SET status = p_status WHERE task_id = p_task_id;
END;
/
-- Set or upsert a monthly budget for a user
CREATE OR REPLACE PROCEDURE set_budget (
  p_user_id IN NUMBER,
  p_month   IN VARCHAR2,
  p_amount  IN NUMBER
) AS
  v_exists NUMBER;
BEGIN
  SELECT COUNT(*) INTO v_exists FROM budgets WHERE user_id = p_user_id AND month = p_month;
  IF v_exists = 0 THEN
    INSERT INTO budgets (user_id, month, amount) VALUES (p_user_id, p_month, p_amount);
  ELSE
    UPDATE budgets SET amount = p_amount WHERE user_id = p_user_id AND month = p_month;
  END IF;
END;
/
-- Function: remaining budget for a user & month (uses expenses by month text match)
CREATE OR REPLACE FUNCTION get_remaining_budget (
  p_user_id IN NUMBER,
  p_month   IN VARCHAR2   -- e.g., 'August 2025'
) RETURN NUMBER IS
  v_budget   NUMBER := 0;
  v_spent    NUMBER := 0;
BEGIN
  SELECT NVL(SUM(b.amount),0) INTO v_budget
  FROM budgets b
  WHERE b.user_id = p_user_id AND b.month = p_month;

  -- Match by month/year text; you can adapt to a proper month table if needed
  SELECT NVL(SUM(e.amount),0) INTO v_spent
  FROM expenses e
  WHERE e.user_id = p_user_id
    AND TO_CHAR(e.expense_date,'Month YYYY') = p_month;

  RETURN v_budget - v_spent;
END;
