DROP TABLE expenses CASCADE CONSTRAINTS;
DROP TABLE tasks CASCADE CONSTRAINTS;
DROP TABLE budgets CASCADE CONSTRAINTS;
DROP TABLE categories CASCADE CONSTRAINTS;
DROP TABLE users CASCADE CONSTRAINTS;

-- Users table
CREATE TABLE users (
    user_id INT PRIMARY KEY,
    username VARCHAR2(50) NOT NULL,
    email VARCHAR2(100) NOT NULL UNIQUE,
    password VARCHAR2(50) NOT NULL
);

-- Categories table
CREATE TABLE categories (
    category_id INT PRIMARY KEY,
    category_name VARCHAR2(50) NOT NULL,
    category_type VARCHAR2(10) CHECK (category_type IN ('task','expense'))
);

-- Tasks table
CREATE TABLE tasks (
    task_id INT PRIMARY KEY,
    user_id INT NOT NULL,
    title VARCHAR2(100) NOT NULL,
    description VARCHAR2(50),
    category_id INT,
    status VARCHAR2(10) DEFAULT 'pending' CHECK (status IN ('pending','completed')),
    due_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (category_id) REFERENCES categories(category_id)
);

-- Expenses table
CREATE TABLE expenses (
    expense_id INT PRIMARY KEY,
    user_id INT NOT NULL,
    category_id INT NOT NULL,
    task_id INT NULL,
    amount NUMBER(10,2) NOT NULL,
    description VARCHAR2(50),
    expense_date DATE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(user_id),
    FOREIGN KEY (category_id) REFERENCES categories(category_id),
    FOREIGN KEY (task_id) REFERENCES tasks(task_id)
);

-- Budgets table
CREATE TABLE budgets (
    budget_id INT PRIMARY KEY,
    user_id INT NOT NULL,
    month VARCHAR2(20) NOT NULL,
    amount NUMBER(10,2) NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(user_id)
);
